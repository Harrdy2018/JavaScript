### 查找一个数在数组中的位置
```node
//遍历 时间复杂度O(n)
var list = [1,5,7,8,11,23,54];
function findNum1(arr,num){
  for(let i=0;i<arr.length;i++){
    if(arr[i]===num){
      return i;
    }
  }
  return -1;
}
console.log(111,findNum1(list,23))
console.log(222,findNum1(list,88))

//二分查找 时间复杂度O(logn) 线性递增或者递减数组
function findNum2(arr,num){
  var max=arr.length-1;
  var min=0;
  do{
    var mid=parseInt((min+max)/2)
    if(arr[mid]>num){
      max=mid-1
    }else if(arr[mid]<num){
      min=mid+1
    }else{
      return mid
    }
  }while(min<=max)
  return -1;
}
console.log(333,findNum2(list,23))
console.log(444,findNum2(list,88))

//递归查找无序数组中的一个数字
function findNum3(arr,left,right,num){
  var mid=parseInt((left+right)/2)
  if(left===mid || right===mid){
    if(arr[left]===num){
      return left
    }else if(arr[right]===num){
      return right
    }else{
      return -1
    }
  }
  else{
    if(arr.slice(left,mid+1).indexOf(num) !== -1){
      right=mid
      return findNum3(arr,left,right,num)
    }else{
      left=mid
      return findNum3(arr,left,right,num)
    }
  }
}
console.log(555,findNum3(list,0,6,23))
console.log(666,findNum3(list,0,6,88))
```
### 递归
* 求1~100的和
```node
递推关系 sum(n)=sum(n-1)+1
寻找临界点
求100转化为求99
求99转化为求98
...
求3转化为求2
求2转化为求1
即sum(1)=1

function sum(n){
  if(n===1){
    return 1;
  }else{
    return sum(n-1)+n
  }
}
console.log(sum(100))
```
* 求 1,3,5,7,9,...前n项和,序号从0开始
```node
求和递归 sum(n)=sum(n-1)+f(n)
临界点   sum(0)=1

元素递归 f(n)=f(n-1)+2
临界点   f(0)=1

function f(n){
  if(n===0){
    return 1;
  }else{
    return f(n-1)+2
  }
}
function sum(n){
  if(n===0){
    return 1;
  }else{
    return sum(n-1)+f(n)
  }
}
console.log(sum(4))
```

* 数列 1,1,2,4,7,11,16...求前n项和
```node
function f(n){
  if(n===1 || n===2){
    return 1
  }
  return f(n-1)+n-2
}
function sum(n){
  if(n===1){
    return 1
  }else if(n===2){
    return 2
  }
  return sum(n-1)+f(n)
}
[1,2,3,4,5].forEach(i=>{
  console.log(sum(i))
})
```
